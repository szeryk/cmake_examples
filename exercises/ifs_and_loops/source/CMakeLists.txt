cmake_minimum_required(VERSION 3.2)
project(IfsAndLoopsExercise)

# fizzbuzz 
foreach(i RANGE 0 20)
    math(EXPR remainderFrom3 "${i} % 3")
    math(EXPR remainderFrom5 "${i} % 5")
    if((NOT remainderFrom3) AND (NOT remainderFrom5))
        message("FizzBuzz")
    elseif(NOT remainderFrom3)
        message("Fizz")
    elseif(NOT remainderFrom5)
        message("Buzz")
    else()
        message(${i})
    endif()
endforeach()
