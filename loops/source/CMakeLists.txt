cmake_minimum_required(VERSION 3.2)
project(LoopsDemo)


message("Example 1")
foreach(i ONE TWO THREE)
    message(${i})
endforeach()
message("")

message("Example 2")
set(myList Eryk Kasia)
set(yourList Damian Maciek)
foreach(i IN LISTS myList yourList ITEMS Patryk Zosia)
    message(${i})
endforeach()
message("")

message("Example 3")
foreach(i IN ZIP_LISTS myList yourList)
    message(${i_0} " : " ${i_1})
endforeach()
message("")
foreach(i j IN ZIP_LISTS myList yourList)
    message(${i} " : " ${j})
endforeach()
message("")

message("Example 4")
foreach(i RANGE 0 3 1) # stop is inclusive :/
    message(${i})
endforeach()
message("")

message("Example 5")
set(counter 5)
while(YES)
    message(${counter})
    math(EXPR counter "${counter} - 1")
    if(counter EQUAL 2)
        break()
    endif()
endwhile()
message("")
